\begin{align*} T(cf+dg)\amp =(cf+dg)' \amp \text{ (def. of \(T\)) }\\ \amp =(cf)'+(dg)'\amp \text{ (addition prop. of derivative) }\\ \amp =cf'+dg' \amp \text{ (scalar prop. of derivative) }\\ \amp =cT(f)+dT(g) \amp \text{ (def. of \(T\)) } \end{align*}
